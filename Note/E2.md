```
CREATE TABLE Food (
    foodid INT(4) PRIMARY KEY NOT NULL AUTO_INCREMENT,
    Name VARCHAR(20) NOT NULL,
    Company VARCHAR(30) NOT NULL,
    Price FLOAT NOT NULL,
    Product_time YEAR NULL,
    Validity_time INT(4) NULL,
    address VARCHAR(50) NULL
);

```
